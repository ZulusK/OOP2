015    double _delta = 1e-6;
016
017    @Test
018    void constructorProduct() {
019        double minCost = 100;
020        double maxCost = 200;
021        String name = "Pen";
022        Product product = new Product(minCost, maxCost, name);
023        assertAll(
024                () -> assertEquals(product.getMaxCost(), maxCost, _delta),
025                () -> assertEquals(product.getMinCost(), minCost, _delta),
026                () -> assertEquals(product.getName(), name));
027    }
028
029    @Test
030    void constructorProductWithInvalidCosts() {
031        double minCost = 200;
032        double maxCost = 100;
033        String name = "Pen";
034        assertThrows(IllegalArgumentException.class, () -> {
035            Product product = new Product(minCost, maxCost, name);
036        });
037    }
038
039    @Test
040    void constructorManager() {
041        double extraCharge = 0.15;
042        String name = "Sam";
043        Manager M = new Manager(name, extraCharge);
044        assertEquals(M.getExtraCharge(), extraCharge, _delta);
045        assertEquals(M.getName(), name);
046    }
047
048    @Test
049    void constructorManagerWithInvalidCharge() {

066    @Test
067    void SellerUnsupportedOperations() {
068        Seller S1 = new Seller("Bill", 1);
069        Seller S2 = new Seller("Young Tau", 2);
070        assertAll(
071                () -> assertThrows(UnsupportedOperationException.class, () -> S1.add((S2))),
072                () -> assertThrows(UnsupportedOperationException.class, () -> S1.remove((S2)))
073        );

076    @Test

084
085    @Test
086    void ManagerAddEmployee() {
087        Manager M = new Manager("Victor", 1);
088        Seller S1 = new Seller("Rick", 2);
089        Seller S2 = new Seller("Chuck", 2);
090        assertTrue(M.add(S1));
091        assertTrue(M.add(S2));
092        assertFalse(M.add(S1));
093        assertThrows(IllegalArgumentException.class, () -> M.add(null));
094        assertThrows(IllegalArgumentException.class, () -> M.add(M));
095    }
096
097    @Test
098    void ManagerRemoveEmployee() {
099        Manager M = new Manager("Victor", 1);
100        Seller S1 = new Seller("Rick", 2);
101        Seller S2 = new Seller("Chuck", 2);
102        M.add(S1);
103        M.add(S2);
104        assertTrue(M.remove(S1));
105        assertFalse(M.remove(S1));
106        assertThrows(IllegalArgumentException.class, () -> M.remove(null));
107    }
108
109    @Test
110    void ManagerSellProduct() {
111        Manager M = new Manager("Victor", 0.15);
112        Product P = new Product(100, 200, "Book");
113        assertFalse(M.sell(P));
114        M.add(new Seller("Bill", 0.03));
115        System.out.println("=========================================================");
116        assertTrue(M.sell(P));
117        assertThrows(IllegalArgumentException.class, () -> M.sell(null));
118    }

129    @Test
130    void NetworkOfTraderSellProduct() {
131        Product P = new Product(100, 1000, "PC");
132        Manager MainM = new Manager("Harry", 0.1);
133        Manager subM1 = new Manager("Kent", 0.25);
134        Manager subM2 = new Manager("Lucia", 0.25);
135        Seller s1 = new Seller("Dominic", 0.4);
136        Seller s2 = new Seller("Ron", 0.35);
137        Seller s3 = new Seller("Willy", 0.5);
138        MainM.add(subM1);
139        MainM.add(subM2);
140        MainM.add(s1);
141        subM1.add(s2);
142        subM1.add(s3);
143        System.out.println("=========================================================");
144        assertTrue(MainM.sell(P));
145    }
