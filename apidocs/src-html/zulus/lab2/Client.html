007/**
008 * defines Client for financial operations
009 */
010public class Client {
011    String _name;
012    double _money;
013
014    /**
015     * Create new client with specified name and start cash
016     *
017     * @param name name if client
018     * @param cash value cash of client
019     */
020    public Client(String name, double cash) {
021        setName(name);
022        setMoney(cash);
023    }
024
025    /**
026     * get name of client
027     *
028     * @return name of client
029     */
030    public String getName() {
031        return _name;
032    }
033
034    /**
035     * set name of client
036     *
037     * @param name value of name of client
038     */
039    public void setName(String name) {
040        this._name = name;
041    }
042
043    /**
044     * get current cash of client
045     *
046     * @return current cash of client
047     */
048    public double getMoney() {
049        return _money;
050    }
051
052    /**
053     * set value of cash of client
054     *
055     * @param money new value of cash
056     */
057    private void setMoney(double money) {
058        if (money < 0) throw new IllegalArgumentException("Cash must be positive real number");
059        this._money = money;
060    }
061
062    /**
063     * Ask client to pay for product
064     *
065     * @param cost cost of product
066     * @return true, if client had payed for product
067     */
068    public boolean pay(double cost) {
069        if (canPay(cost)) {
070            _money -= cost;
071            return true;
072        } else {
073            return false;
074        }
075    }
076
077    /**
078     * check, can Client pay for product
079     *
080     * @param cost cost of product
081     * @return true, if client can pay for product
082     */
083    public boolean canPay(double cost) {
084        if (cost < 0) throw new IllegalArgumentException("Cost must be positive real number");
085        return cost < _money;
086    }
087
088    @Override
089    public String toString() {
090        return String.format("Client [%s, %.2f]", _name, _money);
091    }
092}
