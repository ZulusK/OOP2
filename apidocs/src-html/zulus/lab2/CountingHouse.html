006public class CountingHouse implements ICountingHouse {
007    String _name;
008
009    public CountingHouse(String name) {
010        setName(name);
011    }
012
013    @Override
014    public String processOperation(double cost, Client client) {
015        client.pay(cost);
016        return String.format("Financial operation[%.2f$] computed for %s", cost, client);
017    }
018
019    public String getName() {
020        return _name;
021    }
022
023    public void setName(String name) {
024        this._name = name;
025    }
026
027    @Override
028    public String toString() {
029        return String.format("Counting house: %s", _name);
030    }
031}
