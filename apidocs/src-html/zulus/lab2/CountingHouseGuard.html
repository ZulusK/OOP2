006public class CountingHouseGuard implements ICountingHouse {
007    ICountingHouse _countingHouse;
008
009    public CountingHouseGuard(ICountingHouse countingHouse) {
010        setCountingHouse(countingHouse);
011    }
012
013    public ICountingHouse getCountinghouse() {
014        return _countingHouse;
015    }
016
017    public void setCountingHouse(ICountingHouse countingHouse) {
018        if (countingHouse == null) throw new IllegalArgumentException("Argument countingHouse must be not-null value");
019        this._countingHouse = countingHouse;
020    }
021
022    @Override
023    public String processOperation(double cost, Client client) {
024        if (client == null) throw new IllegalArgumentException("Argument client must be not-null value");
025        if (!client.canPay(cost))
026            throw new OperationCannotBePerformedException(client, _countingHouse, "Client haven't enough money for this operation");
027        return _countingHouse.processOperation(cost, client);
028    }
029
030}
