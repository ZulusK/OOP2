007public class Seller implements ITradeUnit {
008    double _extraCharge;
009    String _name;
010
011    /**
012     * create new Seller
013     *
014     * @param name        name of seller
015     * @param extraCharge default extraCharge of seller
016     */
017    public Seller(String name, double extraCharge) {
018        setExtraCharge(extraCharge);
019        setName(name);
020    }
021
022    @Override
023    public double getExtraCharge() {
024        return _extraCharge;
025    }
026
027    @Override
028    public void setExtraCharge(double value) {
029        if (value < 0) throw new IllegalArgumentException("Extra charge must be positive real number");
030        this._extraCharge = value;
031    }
032
033    @Override
034    public void setName(String name) {
035        this._name = name;
036    }
037
038    @Override
039    public String getName() {
040        return this._name;
041    }
042
043    @Override
044    public boolean add(ITradeUnit employee) {
045        throw new UnsupportedOperationException("Seller couldn't have employee");
046    }
047
048    @Override
049    public boolean remove(ITradeUnit employee) {
050        throw new UnsupportedOperationException("Seller couldn't have employee");
051    }
052
053    @Override
054    public boolean sell(Product product) {
055        if (product == null)
056            throw new IllegalArgumentException("Argument product must be not-null value");
057        return this.sell(product, product.getMinCost(), 1);
058    }
059
060    @Override
061    public boolean sell(Product product, double costOfProduct, int deep) {
062        if (product == null)
063            throw new IllegalArgumentException("Argument product must be not-null value");
064        if (costOfProduct < product.getMinCost())
065            throw new IllegalArgumentException("CurrentCost must be >= product.minCost");
066
067        double costOfTrade = getCurrentCost(product, costOfProduct);
068        System.out.printf("%" + deep + "s Seller %s sold the `%s` for %.2f$ (+%.2f%%)\n", " ", _name, product, costOfTrade, getCurrentExtraCharge(costOfProduct, costOfTrade));
069        return true;
070    }
071
072    private double getCurrentExtraCharge(double costOfProduct, double costOfTrade) {
073        double gain = costOfTrade - costOfProduct;
074        return gain / 100;
075    }
076
077    private double getCurrentCost(Product P, double currentCost) {
078        return Math.min(P.getMaxCost(), currentCost * (1 + _extraCharge));
079    }
080}
