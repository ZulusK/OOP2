003import java.util.HashSet;
004
005/**
006 * Created by zulus on 21.02.18
007 */
008
009/**
010 * Defines manager of trading company
011 */
012public class Manager implements ITradeUnit {
013    double _extraCharge;
014    String _name;
015    HashSet<ITradeUnit> _employees;
016
017    /**
018     * create new Manager
019     *
020     * @param name        name of manager
021     * @param extraCharge default extraCharge of manager
022     */
023    public Manager(String name, double extraCharge) {
024        setExtraCharge(extraCharge);
025        setName(name);
026        this._employees = new HashSet<>();
027    }
028
029    @Override
030    public double getExtraCharge() {
031        return _extraCharge;
032    }
033
034    @Override
035    public void setExtraCharge(double value) {
036        if (value < 0) throw new IllegalArgumentException("Extra charge must be positive real number");
037        this._extraCharge = value;
038    }
039
040    @Override
041    public void setName(String name) {
042        this._name = name;
043    }
044
045    @Override
046    public String getName() {
047        return this._name;
048    }
049
050    @Override
051    public boolean add(ITradeUnit employee) {
052        if (employee == null) throw new IllegalArgumentException("Argument employee must be not-null value");
053        if (employee == this) throw new IllegalArgumentException("Employee must be not self-pointer");
054        return this._employees.add(employee);
055    }
056
057    @Override
058    public boolean remove(ITradeUnit employee) {
059        if (employee == null) throw new IllegalArgumentException("Argument employee must be not-null value");
060        return this._employees.remove(employee);
061    }
062
063    @Override
064    public boolean sell(Product product) {
065        if (product == null)
066            throw new IllegalArgumentException("Argument product must be not-null value");
067        return this.sell(product, product.getMinCost(), 1);
068    }
069
070    @Override
071    public boolean sell(Product product, double costOfProduct, int deep) {
072        if (product == null)
073            throw new IllegalArgumentException("Argument product must be not-null value");
074        if (costOfProduct < product.getMinCost())
075            throw new IllegalArgumentException("costOfProduct must be >= product.minCost");
076        if (_employees.isEmpty()) {
077            System.out.printf("%" + deep + "s Manager %s cannot sold the `%s` \n", " ", _name, product);
078            return false;
079        } else {
080            double costOfTrade = getCurrentCost(product, costOfProduct);
081            System.out.printf("%" + deep + "s Manager %s sold the `%s` for %.2f$ (+%.2f%%)\n", " ", _name, product, costOfTrade, getCurrentExtraCharge(costOfProduct, costOfTrade));
082            for (ITradeUnit unit : _employees) {
083                unit.sell(product, costOfTrade, deep + 1);
084            }
085            return true;
086        }
087    }
088
089    private double getCurrentExtraCharge(double costOfProduct, double costOfTrade) {
090        double gain = costOfTrade - costOfProduct;
091        return gain / 100;
092    }
093
094    private double getCurrentCost(Product P, double currentCost) {
095        return Math.min(P.getMaxCost(), currentCost * (1 + _extraCharge));
096    }
097}
